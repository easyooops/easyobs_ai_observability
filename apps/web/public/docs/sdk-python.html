Python SDK

Python SDK (easyobs_agent)

A lightweight Python client (~15 KB) that automatically traces all calls from your agent services via OpenTelemetry.

Installation

# Inside the EasyObs project
pip install -e ".[agent]"

# Or from wheel (air-gapped environment)
pip install --no-index --find-links ./deps ./easyobs_agent-0.1.0-py3-none-any.whl

Initialization (init)

Call once at application startup:

from easyobs_agent import init, traced

init(
    "http://<easyobs-server>:8787",
    token="eobs_...",           # Issued from Setup > Organizations > Services
    service="my-agent",         # service.name resource attribute
    auto=True,                  # Enables capture_io + auto_langchain together
)

@traced Decorator

Automatically wraps a function as an OpenTelemetry Span:

from easyobs_agent import traced

@traced                             # span name = function name
def step_a(q: str) -> str:
    return ...

@traced("rag.retrieve")             # fixed span name
def step_b(q: str) -> list[str]:
    return ...

@traced("safe.path", capture=False) # disable arg/return capture
def step_c(secret: str) -> None:
    ...

@traced("force.io", capture=True)   # force capture regardless of init()
def step_d(q: str) -> str:
    return ...

@traced("async.turn")               # async functions supported
async def step_e(q: str) -> str:
    return await llm.ainvoke(q)

span_block (Context Manager)

Create Spans with a with block instead of a decorator:

from easyobs_agent import span_block, record_llm, record_retrieval, record_tool

with span_block("retrieve", kind="retrieve", step="vector.lookup"):
    docs = vector.search(q)
    record_retrieval(query=q, docs=docs)

with span_block("generate", kind="llm"):
    reply = my_local_llm(prompt)
    record_llm(
        model="local-llm-7b",
        query=prompt,
        response=reply,
        tokens_in=128,
        tokens_out=256,
    )

with span_block("postprocess", kind="tool"):
    out = format_markdown(reply)
    record_tool(name="format.markdown", inp=reply, out=out)

span_tag (Attribute Tagging)

Add o.* namespace attributes to the currently active Span:

from easyobs_agent import traced, span_tag, SpanTag

@traced("agent.turn")
def answer(q: str, session_id: str) -> str:
    span_tag(SpanTag.SESSION, session_id)
    span_tag(SpanTag.USER, "alice")
    span_tag(SpanTag.QUERY, q)

    with span_block("retrieve", kind="retrieve"):
        span_tag(SpanTag.TARGET, "faq-index")
        docs = vector.search(q)

    reply = compose(docs)

    span_tag(SpanTag.MODEL, "gpt-4o-mini")
    span_tag(SpanTag.TOKENS_IN, 128)
    span_tag(SpanTag.TOKENS_OUT, 256)
    span_tag(SpanTag.RESPONSE, reply)
    return reply

Helper Functions

record_llm()

Records structured LLM call information on the current span:

record_llm(
    model="gpt-4o-mini",
    query=prompt,
    response=reply,
    tokens_in=128,
    tokens_out=256,
)

record_retrieval()

Records a retrieval/search operation:

record_retrieval(query=q, docs=docs)

record_tool()

Records a tool/function call:

record_tool(name="calculator", inp="2+2", out="4")

record_session()

Sets the session ID for multi-turn tracking:

record_session(session_id="sess_abc123")

Complete Example

from easyobs_agent import init, traced, span_block, record_llm

init("http://localhost:8787", token="eobs_demo_token", service="rag-agent", auto=True)

@traced("agent.turn")
def answer(query: str) -> str:
    with span_block("retrieve", kind="retrieve"):
        docs = search_vector_db(query)

    with span_block("generate", kind="llm"):
        prompt = build_prompt(query, docs)
        response = openai_client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "user", "content": prompt}]
        )
        reply = response.choices[0].message.content
        record_llm(
            model="gpt-4o-mini",
            query=prompt,
            response=reply,
            tokens_in=response.usage.prompt_tokens,
            tokens_out=response.usage.completion_tokens,
        )

    return reply
